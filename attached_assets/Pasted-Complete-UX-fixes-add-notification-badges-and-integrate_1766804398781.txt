```javascript
function setLang(lang) {
  localStorage.setItem('lang', lang);
  if (typeof autoTranslate !== 'undefined') autoTranslate.setLang(lang);
  if (typeof landingTranslate !== 'undefined') landingTranslate.translate();
  if (typeof posthog !== 'undefined') posthog.capture('language_changed', { lang });
  
  document.querySelectorAll('[data-lang-btn]').forEach(btn => {
    const isActive = btn.dataset.langBtn === lang;
    btn.classList.toggle('bg-purple-600', isActive);
    btn.classList.toggle('text-white', isActive);
    btn.classList.toggle('bg-transparent', !isActive);
  });
}

// Initialize on page load
document.addEventListener('DOMContentLoaded', () => {
  const savedLang = localStorage.getItem('lang') || 'en';
  setLang(savedLang);
});
```

### JavaScript to update badges:
```javascript
async function updateSidebarBadges() {
  try {
    // Feedback unread count
    const feedbackRes = await fetch('/api/feedback/unread-count');
    const feedbackData = await feedbackRes.json();
    const feedbackBadge = document.getElementById('feedback-badge');
    if (feedbackBadge) {
      if (feedbackData.count > 0) {
        feedbackBadge.textContent = feedbackData.count > 99 ? '99+' : feedbackData.count;
        feedbackBadge.classList.remove('hidden');
      } else {
        feedbackBadge.classList.add('hidden');
      }
    }

    // Reviews pending count
    const reviewsRes = await fetch('/api/reviews/pending-count');
    const reviewsData = await reviewsRes.json();
    const reviewsBadge = document.getElementById('reviews-badge');
    if (reviewsBadge) {
      if (reviewsData.count > 0) {
        reviewsBadge.textContent = reviewsData.count > 99 ? '99+' : reviewsData.count;
        reviewsBadge.classList.remove('hidden');
      } else {
        reviewsBadge.classList.add('hidden');
      }
    }
  } catch (err) {
    console.error('Badge update error:', err);
  }
}

// Update on load and every 60 seconds
document.addEventListener('DOMContentLoaded', () => {
  updateSidebarBadges();
  setInterval(updateSidebarBadges, 60000);
});
```

### API endpoints (add to routes):
```javascript
// GET /api/feedback/unread-count
router.get('/unread-count', async (req, res) => {
  if (!req.session.userId) return res.json({ count: 0 });
  try {
    const result = await pool.query(
      `SELECT COUNT(*) FROM feedback WHERE user_id = $1 AND (status = 'new' OR status IS NULL)`,
      [req.session.userId]
    );
    res.json({ count: parseInt(result.rows[0].count) || 0 });
  } catch (error) {
    res.json({ count: 0 });
  }
});

// GET /api/reviews/pending-count
router.get('/pending-count', async (req, res) => {
  if (!req.session.userId) return res.json({ count: 0 });
  try {
    const result = await pool.query(
      `SELECT COUNT(*) FROM reviews WHERE user_id = $1 AND ai_response IS NOT NULL AND (approved IS NULL OR approved = false)`,
      [req.session.userId]
    );
    res.json({ count: parseInt(result.rows[0].count) || 0 });
  } catch (error) {
    res.json({ count: 0 });
  }
});
```
